Problem 1: Quadratic Equation

Part a: Write a function to solve quadratic equations

solve_quadratic <- function(a, b, c) {
  discriminant <- b^2 - 4*a*c
  
  if (discriminant < 0) {
    return("No real number solution")
  } else if (discriminant == 0) {
    solution <- -b / (2*a)
    return(solution)
  } else {
    solution1 <- (-b + sqrt(discriminant)) / (2*a)
    solution2 <- (-b - sqrt(discriminant)) / (2*a)
    return(c(solution1, solution2))
  }
}

Problem 2: Classification Boundary Point

get_boundary_point <- function(data, x_var, group_var) {
  # Extract x and group columns
  x <- data[[x_var]]
  group <- data[[group_var]]
  
  # Calculate statistics for each group
  groups <- unique(group)
  mu1 <- mean(x[group == groups[1]], na.rm = TRUE)
  mu2 <- mean(x[group == groups[2]], na.rm = TRUE)
  sigma2_1 <- var(x[group == groups[1]], na.rm = TRUE)
  sigma2_2 <- var(x[group == groups[2]], na.rm = TRUE)
  pi1 <- mean(group == groups[1])
  pi2 <- mean(group == groups[2])
  
  # Calculate coefficients for quadratic equation
  a <- -1/(2*sigma2_1) + 1/(2*sigma2_2)
  b <- mu1/sigma2_1 - mu2/sigma2_2
  c <- -mu1^2/(2*sigma2_1) + mu2^2/(2*sigma2_2) + log((pi1/pi2) * (sqrt(sigma2_2)/sqrt(sigma2_1)))
  
  # Solve quadratic equation
  solutions <- solve_quadratic(a, b, c)
  
  # If no real solutions
  if (is.character(solutions)) {
    return(solutions)
  }
  
  # Select solution within range of x
  x_range <- range(x, na.rm = TRUE)
  valid_solutions <- solutions[solutions >= x_range[1] & solutions <= x_range[2]]
  
  if (length(valid_solutions) == 0) {
    return("No solution within data range")
  } else if (length(valid_solutions) == 1) {
    return(valid_solutions)
  } else {
    # Return the solution closest to the middle of the data
    return(valid_solutions[which.min(abs(valid_solutions - mean(x, na.rm = TRUE)))])
  }
}

Problem 5: Beaufort Scale

# Load weather data
weather <- nycflights13::weather

# Define season
weather <- weather %>%
  mutate(
    season = case_when(
      month %in% c(12, 1, 2) ~ "Winter",
      month %in% c(3, 4, 5) ~ "Spring",
      month %in% c(6, 7, 8) ~ "Summer",
      month %in% c(9, 10, 11) ~ "Fall"
    ),
    season = factor(season, levels = c("Winter", "Spring", "Summer", "Fall"))
  )

# Define wind direction
weather <- weather %>%
  mutate(
    wind_dir_cat = case_when(
      wind_dir >= 337.5 | wind_dir < 22.5 ~ "N",
      between(wind_dir, 22.5, 67.5) ~ "NE",
      between(wind_dir, 67.5, 112.5) ~ "E",
      between(wind_dir, 112.5, 157.5) ~ "SE",
      between(wind_dir, 157.5, 202.5) ~ "S",
      between(wind_dir, 202.5, 247.5) ~ "SW",
      between(wind_dir, 247.5, 292.5) ~ "W",
      between(wind_dir, 292.5, 337.5) ~ "NW",
      TRUE ~ NA_character_
    )
  )

# Handle suspicious wind speed observations
# Check for 0 or very low wind speeds that might be measurement errors
weather <- weather %>%
  mutate(
    wind_speed_clean = if_else(wind_speed == 0 & !is.na(wind_gust) & wind_gust > 0,
                                NA_real_,
                                wind_speed)
  )

# Define Beaufort scale
weather <- weather %>%
  mutate(
    beaufort = case_when(
      is.na(wind_speed_clean) ~ NA_character_,
